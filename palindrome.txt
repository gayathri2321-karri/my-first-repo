    <title>Palindrome Checker</title>

<body>

    <h2>Palindrome Checker</h2>

    <input type="text" id="textInput" placeholder="Enter a word">
    <button onclick="checkPalindrome()">Check</button>

    <p id="result"></p>

    <script>
        function checkPalindrome() {
            let str = document.getElementById("textInput").value;
            let reversed = str.split("").reverse().join("");

            if (str === reversed && str !== "") {
                document.getElementById("result").innerHTML = "It is a Palindrome!";
            } else {
                document.getElementById("result").innerHTML = "Not a Palindrome.";
            }
        }
    </script>
